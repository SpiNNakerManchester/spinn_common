519 static int64_t __stdfix_stochastic_round_s64(
525  int64_t x, int n)
526 {
527  register int64_t r;
528  register uint64_t p, dropped_bits;
529 
530  r = x >> n;
531  p = RANDOM() >> (32 - n);
532  dropped_bits = x & (((uint64_t) 0x1 << n) - 1);
533 
534  if (p < dropped_bits) {
535  return (r + 0x1) << n;
536  } else {
537  return r << n;
538  }
539 }

546 static int64_t __stdfix_stochastic_round_s64_alternative(
547  int64_t x, int n)
548 {
549  register int64_t r;
550  register uint64_t p;
551 
552  p = RANDOM() & (((uint64_t) 0x1 << n) - 1);
553  r = x + p;
554  return r;
555 }

678  int32_t x, int32_t y)
679 {
