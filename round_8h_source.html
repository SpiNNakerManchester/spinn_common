529 static int64_t __stdfix_stochastic_round_s64(int64_t x, int n)
535 {
536  register int64_t r;
537  register uint64_t p, dropped_bits;
538 
539  r = x >> n;
540  p = RANDOM() >> (32 - n);
541  dropped_bits = x & (((uint64_t) 0x1 << n) - 1);
542 
543  if (p < dropped_bits) {
544  return (r + 0x1) << n;
545  } else {
546  return r << n;
547  }
548 }

556 {
557  register int64_t r;
558  register uint64_t p;
559 
560  p = RANDOM() & (((uint64_t) 0x1 << n) - 1);
561  r = x + p;
562  return r;
563 }
