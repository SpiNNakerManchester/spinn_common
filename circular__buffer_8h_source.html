80static inline bool circular_buffer_add(circular_buffer buffer, uint32_t item) {
81 uint32_t next = _circular_buffer_next(buffer, buffer->input);
82 bool success = _circular_buffer_not_full(buffer, next);
83
84 if (success) {
85 buffer->buffer[buffer->input] = item;
86 buffer->input = next;
87 } else {
88 buffer->overflows++;
89 }
90
91 return success;
92}

98static inline bool circular_buffer_get_next(
99 circular_buffer buffer, uint32_t *item) {
100 bool success = _circular_buffer_not_empty(buffer);
101
102 if (success) {
103 *item = buffer->buffer[buffer->output];
104 buffer->output = _circular_buffer_next(buffer, buffer->output);
105 }
106
107 return success;
108}

114static inline bool circular_buffer_advance_if_next_equals(
115 circular_buffer buffer, uint32_t item) {
116 bool success = _circular_buffer_not_empty(buffer);
117 if (success) {
118 success = (buffer->buffer[buffer->output] == item);
119 if (success) {
120 buffer->output = _circular_buffer_next(buffer, buffer->output);
121 }
122 }
123 return success;
124}
