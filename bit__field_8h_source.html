229 static inline size_t get_bit_field_size(
230  size_t bits)
231 {
232  // **NOTE** in floating point terms this is ceil(num_neurons / 32)
233  const uint32_t bits_to_words_shift = 5;
234  const uint32_t bits_to_words_remainder = (1 << bits_to_words_shift) - 1;
235 
236  // Down shift number of bits to words
237  uint32_t words = bits >> bits_to_words_shift;
238 
239  // If there was a remainder, add an extra word
240  if ((bits & bits_to_words_remainder) != 0) {
241  words++;
242  }
243  return words;
244 }
